# 栈

[leetcode20 有效的括号](https://leetcode-cn.com/problems/valid-parentheses/)

- 利用栈的特性进行计算

```js
var isValid = function(s) {
  const n = s.length;
  if (n % 2 === 1) {
    return false;
  }
  const map = {
    '}': '{',
    ']': '[',
    ')': '(',
  };
  const stack = [];
  s.split('').forEach(item => {
    if (map[item]) {
      if (stack.length !== 0 && stack[stack.length - 1] === map[item]) {
        stack.pop();
      } else {
        stack.push(item);
      }
    } else {
      stack.push(item);
    }
  });
  return !stack.length;
};
```
